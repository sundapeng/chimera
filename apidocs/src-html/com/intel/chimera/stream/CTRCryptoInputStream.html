039/**
040 * CTRCryptoInputStream decrypts data. AES CTR mode is required in order to
041 * ensure that the plain text and cipher text have a 1:1 mapping. CTR crypto
042 * stream has stream characteristic which is useful for implement features
043 * like random seek. The decryption is buffer based. The key points of the
044 * decryption are (1) calculating the counter and (2) padding through stream
045 * position:
046 * <p/>
047 * counter = base + pos/(algorithm blocksize);
048 * padding = pos%(algorithm blocksize);
049 * <p/>
050 * The underlying stream offset is maintained as state. It is not thread-safe.
051 */
052public class CTRCryptoInputStream extends CryptoInputStream {
053  /**
054   * Underlying stream offset
055   */
056  protected long streamOffset = 0;
057
058  /**
059   * Padding = pos%(algorithm blocksize); Padding is put into {@link #inBuffer}
060   * before any other data goes in. The purpose of padding is to put the input
061   * data at proper position.
062   */
063  private byte padding;
064
065  /**
066   * Flag to mark whether the cipher has been reset
067   */
068  private boolean cipherReset = false;

125  public CTRCryptoInputStream(
126      Input input,
127      Cipher cipher,
128      int bufferSize,
129      byte[] key,
130      byte[] iv,
131      long streamOffset) throws IOException {
132    super(input, cipher, bufferSize, key, iv);

134    Utils.checkStreamCipher(cipher);
135
136    resetStreamOffset(streamOffset);
137  }

139  /**
140   * Decryption is buffer based.
141   * If there is data in {@link #outBuffer}, then read it out of this buffer.
142   * If there is no data in {@link #outBuffer}, then read more from the
143   * underlying stream and do the decryption.
144   * @param b the buffer into which the decrypted data is read.
145   * @param off the buffer offset.
146   * @param len the maximum number of decrypted data bytes to read.
147   * @return int the total number of decrypted data bytes read into the buffer.
148   * @throws IOException
149   */
150  @Override
151  public int read(byte[] b, int off, int len) throws IOException {
152    checkStream();
153    if (b == null) {
154      throw new NullPointerException();
155    } else if (off < 0 || len < 0 || len > b.length - off) {
156      throw new IndexOutOfBoundsException();
157    } else if (len == 0) {
158      return 0;
159    }
160
161    final int remaining = outBuffer.remaining();
162    if (remaining > 0) {
163      int n = Math.min(len, remaining);
164      outBuffer.get(b, off, n);
165      return n;
166    } else {
167      int nd = decryptMore();
168      if(nd <= 0)
169        return nd;
170
171      int n = Math.min(len, outBuffer.remaining());
172      outBuffer.get(b, off, n);
173      return n;
174    }
175  }
176
177  /** Skip n bytes */
178  @Override
179  public long skip(long n) throws IOException {
180    Utils.checkArgument(n >= 0, "Negative skip length.");
181    checkStream();
182
183    if (n == 0) {
184      return 0;
185    } else if (n <= outBuffer.remaining()) {
186      int pos = outBuffer.position() + (int) n;
187      outBuffer.position(pos);
188      return n;
189    } else {
190      /*
191       * Subtract outBuffer.remaining() to see how many bytes we need to
192       * skip in the underlying stream. Add outBuffer.remaining() to the
193       * actual number of skipped bytes in the underlying stream to get the
194       * number of skipped bytes from the user's point of view.
195       */
196      n -= outBuffer.remaining();
197      long skipped = input.skip(n);
198      if (skipped < 0) {
199        skipped = 0;
200      }
201      long pos = streamOffset + skipped;
202      skipped += outBuffer.remaining();
203      resetStreamOffset(pos);
204      return skipped;
205    }
206  }
207
208  /** ByteBuffer read. */
209  @Override
210  public int read(ByteBuffer buf) throws IOException {
211    checkStream();
212    int unread = outBuffer.remaining();
213    if (unread <= 0) { // Fill the unread decrypted data buffer firstly
214      final int n = input.read(inBuffer);
215      if (n <= 0) {
216        return n;
217      }
218
219      streamOffset += n; // Read n bytes
220      if (buf.isDirect() && buf.remaining() >= inBuffer.position() && padding == 0) {
221        // Use buf as the output buffer directly
222        decryptInPlace(buf);
223        padding = postDecryption(streamOffset);
224        return n;
225      } else {
226        // Use outBuffer as the output buffer
227        decrypt();
228        padding = postDecryption(streamOffset);
229      }
230    }
231
232    // Copy decrypted data from outBuffer to buf
233    unread = outBuffer.remaining();
234    final int toRead = buf.remaining();
235    if (toRead <= unread) {
236      final int limit = outBuffer.limit();
237      outBuffer.limit(outBuffer.position() + toRead);
238      buf.put(outBuffer);
239      outBuffer.limit(limit);
240      return toRead;
241    } else {
242      buf.put(outBuffer);
243      return unread;
244    }
245  }
246
247  protected long getStreamOffset() {
248    return streamOffset;
249  }
250
251  /**
252   * Decrypt more data by reading the under layer stream. The decrypted data will
253   * be put in the output buffer.
254   *
255   * @return The number of decrypted data. -1 if end of the decrypted stream
256   */
257  protected int decryptMore() throws IOException {
258    int n = input.read(inBuffer);
259    if (n <= 0) {
260      return n;
261    }
262
263    streamOffset += n; // Read n bytes
264    decrypt();
265    padding = postDecryption(streamOffset);
266    return outBuffer.remaining();
267  }
268
269  /**
270   * Do the decryption using inBuffer as input and outBuffer as output.
271   * Upon return, inBuffer is cleared; the decrypted data starts at
272   * outBuffer.position() and ends at outBuffer.limit();
273   */
274  protected void decrypt() throws IOException {
275    Utils.checkState(inBuffer.position() >= padding);
276    if(inBuffer.position() == padding) {
277      // There is no real data in inBuffer.
278      return;
279    }
280
281    inBuffer.flip();
282    outBuffer.clear();
283    decryptBuffer(outBuffer);
284    inBuffer.clear();
285    outBuffer.flip();
286
287    if (padding > 0) {
288      /*
289       * The plain text and cipher text have a 1:1 mapping, they start at the
290       * same position.
291       */
292      outBuffer.position(padding);
293    }
294  }
295
296  /**
297   * Do the decryption using inBuffer as input and buf as output.
298   * Upon return, inBuffer is cleared; the buf's position will be equal to
299   * <i>p</i>&nbsp;<tt>+</tt>&nbsp;<i>n</i> where <i>p</i> is the position before
300   * decryption, <i>n</i> is the number of bytes decrypted.
301   * The buf's limit will not have changed.
302   */
303  protected void decryptInPlace(ByteBuffer buf) throws IOException {
304    Utils.checkState(inBuffer.position() >= padding);
305    Utils.checkState(buf.isDirect());
306    Utils.checkState(buf.remaining() >= inBuffer.position());
307    Utils.checkState(padding == 0);
308
309    if(inBuffer.position() == padding) {
310      // There is no real data in inBuffer.
311      return;
312    }
313    inBuffer.flip();
314    decryptBuffer(buf);
315    inBuffer.clear();
316  }
317
318  /**
319   * Decrypt all data in buf: total n bytes from given start position.
320   * Output is also buf and same start position.
321   * buf.position() and buf.limit() should be unchanged after decryption.
322   */
323  protected void decrypt(ByteBuffer buf, int offset, int len)
324      throws IOException {
325    final int pos = buf.position();
326    final int limit = buf.limit();
327    int n = 0;
328    while (n < len) {
329      buf.position(offset + n);
330      buf.limit(offset + n + Math.min(len - n, inBuffer.remaining()));
331      inBuffer.put(buf);
332      // Do decryption
333      try {
334        decrypt();
335        buf.position(offset + n);
336        buf.limit(limit);
337        n += outBuffer.remaining();
338        buf.put(outBuffer);
339      } finally {
340        padding = postDecryption(streamOffset - (len - n));
341      }
342    }
343    buf.position(pos);
344  }
345
346  /**
347   * This method is executed immediately after decryption. Check whether
348   * cipher should be updated and recalculate padding if needed.
349   */
350  protected byte postDecryption(long position) throws IOException {
351    byte padding = 0;
352    if (cipherReset) {
353      /*
354       * This code is generally not executed since the cipher usually
355       * maintains cipher context (e.g. the counter) internally. However,
356       * some implementations can't maintain context so a re-init is necessary
357       * after each decryption call.
358       */
359      resetCipher(position);
360      padding = getPadding(position);
361      inBuffer.position(padding);
362    }
363    return padding;
364  }
365
366  protected long getCounter(long position) {
367    return position / cipher.getTransformation().getAlgorithmBlockSize();
368  }
369
370  protected byte getPadding(long position) {
371    return (byte)(position % cipher.getTransformation().getAlgorithmBlockSize());
372  }
373
374  /** Initialize the cipher. */
375  @Override
376  protected void initCipher() {
377    // Do nothing for initCipher
378    // Will reset the cipher when reset the stream offset
379  }
380
381  /** Calculate the counter and iv, reset the cipher. */
382  protected void resetCipher(long position)
383      throws IOException {
384    final long counter = getCounter(position);
385    Utils.calculateIV(initIV, counter, iv);
386    try {
387      cipher.init(Cipher.DECRYPT_MODE, key, iv);
388    } catch (InvalidKeyException e) {
389      throw new IOException(e);
390    } catch (InvalidAlgorithmParameterException e) {
391      throw new IOException(e);
392    }
393    cipherReset = false;
394  }
395
396  /**
397   * Reset the underlying stream offset; clear {@link #inBuffer} and
398   * {@link #outBuffer}. This Typically happens during {@link #skip(long)}.
399   */
400  protected void resetStreamOffset(long offset) throws IOException {
401    streamOffset = offset;
402    inBuffer.clear();
403    outBuffer.clear();
404    outBuffer.limit(0);
405    resetCipher(offset);
406    padding = getPadding(offset);
407    inBuffer.position(padding); // Set proper position for input data.
408  }
409
410  protected void decryptBuffer(ByteBuffer out)
411      throws IOException {
412    int inputSize = inBuffer.remaining();
413    try {
414      int n = cipher.update(inBuffer, out);
415      if (n < inputSize) {
416        /**
417         * Typically code will not get here. Cipher#update will consume all
418         * input data and put result in outBuffer.
419         * Cipher#doFinal will reset the cipher context.
420         */
421        cipher.doFinal(inBuffer, out);
422        cipherReset = true;
423      }
424    } catch (ShortBufferException e) {
425      throw new IOException(e);
426    } catch (IllegalBlockSizeException e) {
427      throw new IOException(e);
428    } catch (BadPaddingException e) {
429      throw new IOException(e);
430    }
431  }
