020import java.nio.ByteBuffer;
021import java.security.NoSuchAlgorithmException;
022import java.util.StringTokenizer;
023
024import javax.crypto.BadPaddingException;
025import javax.crypto.IllegalBlockSizeException;
026import javax.crypto.NoSuchPaddingException;
027import javax.crypto.ShortBufferException;
028
029import org.apache.commons.logging.Log;
030import org.apache.commons.logging.LogFactory;
031
032import com.intel.chimera.utils.NativeCodeLoader;
033import com.intel.chimera.utils.Utils;

035/**
036 * OpenSSL cryptographic wrapper using JNI.
037 * Currently only AES-CTR is supported. It's flexible to add
038 * other crypto algorithms/modes.
039 */
040public final class Openssl {
041  private static final Log LOG = LogFactory.getLog(Openssl.class.getName());
042
043  // Mode constant defined by Openssl JNI
044  public static final int ENCRYPT_MODE = 1;
045  public static final int DECRYPT_MODE = 0;
046
047  /** Currently only support AES/CTR/NoPadding. */
048  private static enum AlgorithmMode {
049    AES_CTR,
050    AES_CBC;
051
052    static int get(String algorithm, String mode)
053        throws NoSuchAlgorithmException {
054      try {
055        return AlgorithmMode.valueOf(algorithm + "_" + mode).ordinal();
056      } catch (Exception e) {
057        throw new NoSuchAlgorithmException("Doesn't support algorithm: " +
058            algorithm + " and mode: " + mode);
059      }
060    }
061  }
062
063  private static enum Padding {
064    NoPadding,
065    PKCS5Padding;
066
067    static int get(String padding) throws NoSuchPaddingException {
068      try {
069        return Padding.valueOf(padding).ordinal();
070      } catch (Exception e) {
071        throw new NoSuchPaddingException("Doesn't support padding: " + padding);
072      }
073    }
074  }
075
076  private long context = 0;
077  private final int algorithm;
078  private final int padding;
079
080  private static final String loadingFailureReason;
081
082  static {
083    String loadingFailure = null;
084    try {
085      if (NativeCodeLoader.isNativeCodeLoaded()) {
086        OpensslNative.initIDs();
087      }
088    } catch (Throwable t) {
089      loadingFailure = t.getMessage();
090      LOG.debug("Failed to load OpenSSL Cipher.", t);
091    } finally {
092      loadingFailureReason = loadingFailure;
093    }
094  }
095
096  public static String getLoadingFailureReason() {
097    return loadingFailureReason;
098  }
099
100  private Openssl(long context, int algorithm, int padding) {
101    this.context = context;
102    this.algorithm = algorithm;
103    this.padding = padding;
104  }
105
106  /**
107   * Return an <code>OpensslCipher<code> object that implements the specified
108   * transformation.
109   *
110   * @param transformation the name of the transformation, e.g.,
111   * AES/CTR/NoPadding.
112   * @return OpensslCipher an <code>OpensslCipher<code> object
113   * @throws NoSuchAlgorithmException if <code>transformation</code> is null,
114   * empty, in an invalid format, or if Openssl doesn't implement the
115   * specified algorithm.
116   * @throws NoSuchPaddingException if <code>transformation</code> contains
117   * a padding scheme that is not available.
118   */
119  public static final Openssl getInstance(String transformation)
120      throws NoSuchAlgorithmException, NoSuchPaddingException {
121    Transform transform = tokenizeTransformation(transformation);
122    int algorithmMode = AlgorithmMode.get(transform.algorithm, transform.mode);
123    int padding = Padding.get(transform.padding);
124    long context = OpensslNative.initContext(algorithmMode, padding);
125    return new Openssl(context, algorithmMode, padding);
126  }
127
128  /** Nested class for algorithm, mode and padding. */
129  private static class Transform {
130    final String algorithm;
131    final String mode;
132    final String padding;
133
134    public Transform(String algorithm, String mode, String padding) {
135      this.algorithm = algorithm;
136      this.mode = mode;
137      this.padding = padding;
138    }
139  }
140
141  private static Transform tokenizeTransformation(String transformation)
142      throws NoSuchAlgorithmException {
143    if (transformation == null) {
144      throw new NoSuchAlgorithmException("No transformation given.");
145    }
146
147    /*
148     * Array containing the components of a Cipher transformation:
149     *
150     * index 0: algorithm (e.g., AES)
151     * index 1: mode (e.g., CTR)
152     * index 2: padding (e.g., NoPadding)
153     */
154    String[] parts = new String[3];
155    int count = 0;
156    StringTokenizer parser = new StringTokenizer(transformation, "/");
157    while (parser.hasMoreTokens() && count < 3) {
158      parts[count++] = parser.nextToken().trim();
159    }
160    if (count != 3 || parser.hasMoreTokens()) {
161      throw new NoSuchAlgorithmException("Invalid transformation format: " +
162          transformation);
163    }
164    return new Transform(parts[0], parts[1], parts[2]);
165  }
166
167  /**
168   * Initialize this cipher with a key and IV.
169   *
170   * @param mode {@link #ENCRYPT_MODE} or {@link #DECRYPT_MODE}
171   * @param key crypto key
172   * @param iv crypto iv
173   */
174  public void init(int mode, byte[] key, byte[] iv) {
175    context = OpensslNative.init(context, mode, algorithm, padding, key, iv);
176  }
177
178  /**
179   * Continues a multiple-part encryption or decryption operation. The data
180   * is encrypted or decrypted, depending on how this cipher was initialized.
181   * <p/>
182   *
183   * All <code>input.remaining()</code> bytes starting at
184   * <code>input.position()</code> are processed. The result is stored in
185   * the output buffer.
186   * <p/>
187   *
188   * Upon return, the input buffer's position will be equal to its limit;
189   * its limit will not have changed. The output buffer's position will have
190   * advanced by n, when n is the value returned by this method; the output
191   * buffer's limit will not have changed.
192   * <p/>
193   *
194   * If <code>output.remaining()</code> bytes are insufficient to hold the
195   * result, a <code>ShortBufferException</code> is thrown.
196   *
197   * @param input the input ByteBuffer
198   * @param output the output ByteBuffer
199   * @return int number of bytes stored in <code>output</code>
200   * @throws ShortBufferException if there is insufficient space in the
201   * output buffer
202   */
203  public int update(ByteBuffer input, ByteBuffer output)
204      throws ShortBufferException {
205    checkState();
206    Utils.checkArgument(input.isDirect() && output.isDirect(),
207        "Direct buffers are required.");
208    int len = OpensslNative.update(context, input, input.position(),
209        input.remaining(), output, output.position(), output.remaining());
210    input.position(input.limit());
211    output.position(output.position() + len);
212    return len;
213  }
214
215  /**
216   * Continues a multiple-part encryption/decryption operation. The data
217   * is encrypted or decrypted, depending on how this cipher was initialized.
218   *
219   * @param input the input byte array
220   * @param inputOffset the offset in input where the input starts
221   * @param inputLen the input length
222   * @param output the byte array for the result
223   * @param outputOffset the offset in output where the result is stored
224   * @return the number of bytes stored in output
225   * @throws ShortBufferException if there is insufficient space in the output byte array
226   */
227  public int update(byte[] input, int inputOffset, int inputLen,
228      byte[] output, int outputOffset)
229      throws ShortBufferException {
230    checkState();
231    return OpensslNative.updateByteArray(context, input, inputOffset, inputLen,
232        output, outputOffset, output.length - outputOffset);
233  }
234
235  /**
236   * Finishes a multiple-part operation. The data is encrypted or decrypted,
237   * depending on how this cipher was initialized.
238   * <p/>
239   *
240   * The result is stored in the output buffer. Upon return, the output buffer's
241   * position will have advanced by n, where n is the value returned by this
242   * method; the output buffer's limit will not have changed.
243   * <p/>
244   *
245   * If <code>output.remaining()</code> bytes are insufficient to hold the result,
246   * a <code>ShortBufferException</code> is thrown.
247   * <p/>
248   *
249   * Upon finishing, this method resets this cipher object to the state it was
250   * in when previously initialized. That is, the object is available to encrypt
251   * or decrypt more data.
252   * <p/>
253   *
254   * If any exception is thrown, this cipher object need to be reset before it
255   * can be used again.
256   *
257   * @param output the output ByteBuffer
258   * @return int number of bytes stored in <code>output</code>
259   * @throws ShortBufferException
260   * @throws IllegalBlockSizeException
261   * @throws BadPaddingException
262   */
263  public int doFinal(ByteBuffer output)
264      throws ShortBufferException, IllegalBlockSizeException,
265      BadPaddingException {
266    checkState();
267    Utils.checkArgument(output.isDirect(), "Direct buffer is required.");
268    int len = OpensslNative.doFinal(context, output, output.position(), output.remaining());
269    output.position(output.position() + len);
270    return len;
271  }
272
273  /**
274   * Encrypts or decrypts data in a single-part operation, or finishes a
275   * multiple-part operation.
276   *
277   * @param output the byte array for the result
278   * @param outputOffset the offset in output where the result is stored
279   * @return the number of bytes stored in output
280   * @throws ShortBufferException if the given output byte array is too small
281   * to hold the result
282   * @throws BadPaddingException if this cipher is in decryption mode,
283   * and (un)padding has been requested, but the decrypted data is not
284   * bounded by the appropriate padding bytes
285   * @throws IllegalBlockSizeException if this cipher is a block cipher,
286   * no padding has been requested (only in encryption mode), and the total
287   * input length of the data processed by this cipher is not a multiple of
288   * block size; or if this encryption algorithm is unable to
289   * process the input data provided.
290   */
291  public int doFinal(byte[] output, int outputOffset)
292      throws ShortBufferException, IllegalBlockSizeException, BadPaddingException {
293    checkState();
294    return OpensslNative.doFinalByteArray(context,
295        output, outputOffset, output.length - outputOffset);
296  }
297
298  /** Forcibly clean the context. */
299  public void clean() {
300    if (context != 0) {
301      OpensslNative.clean(context);
302      context = 0;
303    }
304  }
305
306  /** Check whether context is initialized. */
307  private void checkState() {
308    Utils.checkState(context != 0);
309  }
310
311  @Override
312  protected void finalize() throws Throwable {
313    clean();
314  }
