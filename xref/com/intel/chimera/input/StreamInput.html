18  package com.intel.chimera.input;
19  
20  import java.io.IOException;
21  import java.io.InputStream;
22  import java.nio.ByteBuffer;
23  
24  /**
25   * The StreamInput class takes a <code>InputStream</code> object and
26   * wraps it as <code>Input</code> object acceptable by <code>CryptoInputStream</code>.
27   */
28  public class StreamInput implements Input {
29    private byte[] buf;
30    private int bufferSize;
31    InputStream in;
32  
33    public StreamInput(InputStream inputStream, int bufferSize) {
34      this.in = inputStream;
35      this.bufferSize = bufferSize;
36    }
37  
38    @Override
39    public int read(ByteBuffer dst) throws IOException {
40      int remaining = dst.remaining();
41      final byte[] tmp = getBuf();
42      int read = 0;
43      while (remaining > 0) {
44        final int n = in.read(tmp, 0, Math.min(remaining, bufferSize));
45        if (n == -1) {
46          if (read == 0) {
47            read = -1;
48          }
49          break;
50        } else if (n > 0) {
51          dst.put(tmp, 0, n);
52          read += n;
53          remaining -= n;
54        }
55      }
56      return read;
57    }
58  
59    @Override
60    public long skip(long n) throws IOException {
61      return in.skip(n);
62    }
63  
64    @Override
65    public int available() throws IOException {
66      return in.available();
67    }
68  
69    @Override
70    public int read(long position, byte[] buffer, int offset, int length)
71        throws IOException {
72      throw new UnsupportedOperationException(
73          "Positioned read is not supported by this implementation");
74    }
75  
76    @Override
77    public void readFully(long position, byte[] buffer, int offset, int length)
78        throws IOException {
79      throw new UnsupportedOperationException(
80          "Positioned readFully is not supported by this implementation");
81    }
82  
83    @Override
84    public void seek(long pos) throws IOException {
85      throw new UnsupportedOperationException(
86          "Seek is not supported by this implementation");
87    }
88  
89    @Override
90    public void close() throws IOException {
91      in.close();
92    }
93  
94    private byte[] getBuf() {
95      if (buf == null) {
96        buf = new byte[bufferSize];
97      }
98      return buf;
99    }
100 }
